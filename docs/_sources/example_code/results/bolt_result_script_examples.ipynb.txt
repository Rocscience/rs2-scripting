    "\n",
    "for stageNum, allBoltResults in results.items():\n",

    "    for singleBoltResult in allBoltResults:\n",

    "        start_x = singleBoltResult.start_x\n",
    "        start_y = singleBoltResult.start_y\n",
    "        end_x = singleBoltResult.end_x\n",
